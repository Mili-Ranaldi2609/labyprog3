function saludarUser() {
  const nombre =prompt("¿Cual es tu nombre?") // Obtenemos el valor del campo en la función
  if (nombre.trim() === "") {
    console.log("Ingrese su nombre");
  } else {
    console.log("Hola, ", nombre);
  }
}
